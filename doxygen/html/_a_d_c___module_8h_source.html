163 /* MK20DX256 Datasheet:
164 The 16-bit accuracy specifications listed in Table 24 and Table 25 are achievable on the
165 differential pins ADCx_DP0, ADCx_DM0
166 All other ADC channels meet the 13-bit differential/12-bit single-ended accuracy
167 specifications.
168 
169 The results in this data sheet were derived from a system which has < 8 Ohm analog source resistance. The RAS/CAS
170 time constant should be kept to < 1ns.
171 
172 ADC clock should be 2 to 12 MHz for 16 bit mode
173 ADC clock should be 1 to 18 MHz for 8-12 bit mode, and 1-24 MHz for Teensy 3.5 (NOT 3.4)
174 To use the maximum ADC conversion clock frequency, the ADHSC bit must be set and the ADLPC bit must be clear
175 
176 The ADHSC bit is used to configure a higher clock input frequency. This will allow
177 faster overall conversion times. To meet internal ADC timing requirements, the ADHSC
178 bit adds additional ADCK cycles. Conversions with ADHSC = 1 take two more ADCK
179 cycles. ADHSC should be used when the ADCLK exceeds the limit for ADHSC = 0.
180 
181 */

312 // Settings for the power/speed of conversions/sampling
313 /* For conversion speeds:
314  ADC_VERY_LOW_SPEED is guaranteed to be the lowest possible speed within specs for resolutions less than 16 bits (higher than 1 MHz),
315  it's different from ADC_LOW_SPEED only for 24, 4 or 2 MHz.
316  ADC_LOW_SPEED is guaranteed to be the lowest possible speed within specs for all resolutions (higher than 2 MHz).
317  ADC_MED_SPEED is always >= ADC_LOW_SPEED and <= ADC_HIGH_SPEED.
318  ADC_HIGH_SPEED_16BITS is guaranteed to be the highest possible speed within specs for all resolutions (lower or eq than 12 MHz).
319  ADC_HIGH_SPEED is guaranteed to be the highest possible speed within specs for resolutions less than 16 bits (lower or eq than 18 MHz).
320  ADC_VERY_HIGH_SPEED may be out of specs, it's different from ADC_HIGH_SPEED only for 48, 40 or 24 MHz.
321 */
322 /* For sampling speeds:
323  ADC_VERY_LOW_SPEED is the lowest possible sampling speed (+24 ADCK).
324  ADC_LOW_SPEED adds +16 ADCK.
325  ADC_MED_SPEED adds +10 ADCK.
326  ADC_HIGH_SPEED (or ADC_HIGH_SPEED_16BITS) adds +6 ADCK.
327  ADC_VERY_HIGH_SPEED is the highest possible sampling speed (0 ADCK added).
328 */

331 #define ADC_MED_SPEED 2
332 #define ADC_HIGH_SPEED_16BITS 3
333 #define ADC_HIGH_SPEED 4

894  /* Clear bit in address (make it zero), set bit (make it one) or return the value of that bit
895  * We can change this functions depending on the board.
896  * Teensy 3.x use bitband while Teensy LC has a more advanced bit manipulation engine.
897  */
898 #if defined(ADC_TEENSY_3_1) || defined(ADC_TEENSY_3_0) || defined(ADC_TEENSY_3_4) || defined(ADC_TEENSY_3_5)
899  // bitband
900 #define ADC_BITBAND_ADDR(reg, bit) (((uint32_t)(reg) - 0x40000000) * 32 + (bit) * 4 + 0x42000000)
901 
902  __attribute__((always_inline)) void setBit(volatile uint32_t* reg, uint8_t bit) {
903  (*(uint32_t *)ADC_BITBAND_ADDR((reg), (bit))) = 1;
904  }
905  __attribute__((always_inline)) void clearBit(volatile uint32_t* reg, uint8_t bit) {
906  (*(uint32_t *)ADC_BITBAND_ADDR((reg), (bit))) = 0;
907  }
908 
909  __attribute__((always_inline)) void changeBit(volatile uint32_t* reg, uint8_t bit, bool state) {
910  (*(uint32_t *)ADC_BITBAND_ADDR((reg), (bit))) = state;
911  }
912 
913  __attribute__((always_inline)) volatile bool getBit(volatile uint32_t* reg, uint8_t bit) {
914  return (volatile bool)*(uint32_t*)(ADC_BITBAND_ADDR(reg, bit));
915  }
916 
917 #elif defined(ADC_TEENSY_LC)
918  // bit manipulation engine
919  //#define ADC_SETBIT_ATOMIC(reg, bit) (*(uint32_t *)(((uint32_t)&(reg) - 0xF8000000) | 0x48000000) = 1 << (bit)) // OR
920  //#define ADC_CLRBIT_ATOMIC(reg, bit) (*(uint32_t *)(((uint32_t)&(reg) - 0xF8000000) | 0x44000000) = ~(1 << (bit))) // XOR
921 
922  __attribute__((always_inline)) void setBit(volatile uint32_t* reg, uint8_t bit) {
923  //temp = *(uint32_t *)((uint32_t)(reg) | (1<<26) | (bit<<21)); // LAS
924  *(volatile uint32_t*)((uint32_t)(reg) | (1<<27)) = 1<<bit; // OR
925  }
926  __attribute__((always_inline)) void clearBit(volatile uint32_t* reg, uint8_t bit) {
927  //temp = *(uint32_t *)((uint32_t)(reg) | (3<<27) | (bit<<21)); // LAC
928  *(volatile uint32_t*)((uint32_t)(reg) | (1<<26)) = ~(1<<bit); // AND
929  }
930 
931  __attribute__((always_inline)) void changeBit(volatile uint32_t* reg, uint8_t bit, bool state) {
932  //temp = *(uint32_t *)((uint32_t)(reg) | ((3-2*!!state)<<27) | (bit<<21)); // LAS/LAC
933  if(state) { // set
934  *(volatile uint32_t*)((uint32_t)(reg) | (1<<27)) = 1<<bit; // OR
935  } else { // clear
936  *(volatile uint32_t*)((uint32_t)(reg) | (1<<26)) = ~(1<<bit); // AND
937  }
938 
939  }
940 
941  __attribute__((always_inline)) volatile bool getBit(volatile uint32_t* reg, uint8_t bit) {
942  return (volatile bool)*(uint32_t *)((uint32_t)(reg) | (1<<28) | (bit<<23) ); // UBFX
943  }
