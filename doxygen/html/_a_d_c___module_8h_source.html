45 /* MK20DX256 Datasheet:
46 The 16-bit accuracy specifications listed in Table 24 and Table 25 are achievable on the
47 differential pins ADCx_DP0, ADCx_DM0
48 All other ADC channels meet the 13-bit differential/12-bit single-ended accuracy
49 specifications.
50 
51 The results in this data sheet were derived from a system which has < 8 Ohm analog source resistance. The RAS/CAS
52 time constant should be kept to < 1ns.
53 
54 ADC clock should be 2 to 12 MHz for 16 bit mode
55 ADC clock should be 1 to 18 MHz for 8-12 bit mode
56 To use the maximum ADC conversion clock frequency, the ADHSC bit must be set and the ADLPC bit must be clear
57 
58 The ADHSC bit is used to configure a higher clock input frequency. This will allow
59 faster overall conversion times. To meet internal ADC timing requirements, the ADHSC
60 bit adds additional ADCK cycles. Conversions with ADHSC = 1 take two more ADCK
61 cycles. ADHSC should be used when the ADCLK exceeds the limit for ADHSC = 0.

194 // Settings for the power/speed of conversions/sampling
195 /* For conversion speeds:
196  ADC_VERY_LOW_SPEED is guaranteed to be the lowest possible speed within specs for resolutions less than 16 bits (higher than 1 MHz),
197  it's different from ADC_LOW_SPEED only for 24, 4 or 2 MHz.
198  ADC_LOW_SPEED is guaranteed to be the lowest possible speed within specs for all resolutions (higher than 2 MHz).
199  ADC_MED_SPEED is always >= ADC_LOW_SPEED and <= ADC_HIGH_SPEED.
200  ADC_HIGH_SPEED_16BITS is guaranteed to be the highest possible speed within specs for all resolutions (lower or eq than 12 MHz).
201  ADC_HIGH_SPEED is guaranteed to be the highest possible speed within specs for resolutions less than 16 bits (lower or eq than 18 MHz).
202  ADC_VERY_HIGH_SPEED may be out of specs, it's different from ADC_HIGH_SPEED only for 48, 40 or 24 MHz.
203 */
204 /* For sampling speeds:
205  ADC_VERY_LOW_SPEED is the lowest possible sampling speed (+24 ADCK).
206  ADC_LOW_SPEED adds +16 ADCK.
207  ADC_MED_SPEED adds +10 ADCK.
208  ADC_HIGH_SPEED (or ADC_HIGH_SPEED_16BITS) adds +6 ADCK.
209  ADC_VERY_HIGH_SPEED is the highest possible sampling speed (0 ADCK added).
210 */
211 #define ADC_VERY_LOW_SPEED 0
212 #define ADC_LOW_SPEED 1
213 #define ADC_MED_SPEED 2
214 #define ADC_HIGH_SPEED_16BITS 3
215 #define ADC_HIGH_SPEED 4
216 #define ADC_VERY_HIGH_SPEED 5

658  /* Clear bit in address (make it zero), set bit (make it one) or return the value of that bit

660  * Teensy 3.x use bitband while Teensy LC has a more advanced bit manipulation engine.

662  #if defined(ADC_TEENSY_3_1) || defined(ADC_TEENSY_3_0)
663  // bitband
664  #define ADC_BITBAND_ADDR(reg, bit) (((uint32_t)(reg) - 0x40000000) * 32 + (bit) * 4 + 0x42000000)
665 
666  void setBit(volatile uint32_t* reg, uint8_t bit) {
667  (*(uint32_t *)ADC_BITBAND_ADDR((reg), (bit))) = 1;
668  }
