277 /* Returns the analog value of the pin.
278 * It waits until the value is read and then returns the result.
279 * If a comparison has been set up and fails, it will return ADC_ERROR_VALUE.
280 * This function is interrupt safe, so it will restore the adc to the state it was before being called
281 * If more than one ADC exists, it will select the module with less workload, you can force a selection using
282 * adc_num. If you select ADC1 in Teensy 3.0 it will return ADC_ERROR_VALUE.
283 */
284 int ADC::analogRead(uint8_t pin, int8_t adc_num)
285 {
286 #ifdef ADC_SINGLE_ADC
287  return adc0->analogRead(pin); // use ADC0
288 #else
289  /* Teensy 3.1
290  */
291  if (adc_num == -1)
292  { // use no ADC in particular
293  // check which ADC can read the pin
294  bool adc0Pin = adc0->checkPin(pin);
295  bool adc1Pin = adc1->checkPin(pin);
296 
297  if (adc0Pin && adc1Pin)
298  { // Both ADCs
299  if ((adc0->num_measurements) > (adc1->num_measurements))
300  { // use the ADC with less workload
301  return adc1->analogRead(pin);
302  }
303  else
304  {
305  return adc0->analogRead(pin);
306  }
307  }
308  else if (adc0Pin)
309  { // ADC0
310  return adc0->analogRead(pin);
311  }
312  else if (adc1Pin)
313  { // ADC1
314  return adc1->analogRead(pin);
315  }
316  else
317  { // pin not valid in any ADC
318  adc0->fail_flag |= ADC_ERROR::WRONG_PIN;
319  adc1->fail_flag |= ADC_ERROR::WRONG_PIN;
320  return ADC_ERROR_VALUE; // all others are invalid
321  }
322  }
323  else if (adc_num == 0)
324  { // user wants ADC0
325  return adc0->analogRead(pin);
326  }
327  else if (adc_num == 1)
328  { // user wants ADC 1
329  return adc1->analogRead(pin);
330  }
331  adc0->fail_flag |= ADC_ERROR::OTHER;
332  return ADC_ERROR_VALUE;
333 #endif
334 }
335 
336 #if ADC_DIFF_PAIRS > 0
337 /* Reads the differential analog value of two pins (pinP - pinN).
338 * It waits until the value is read and then returns the result.
339 * If a comparison has been set up and fails, it will return ADC_ERROR_VALUE.
340 * \param pinP must be A10 or A12.
341 * \param pinN must be A11 (if pinP=A10) or A13 (if pinP=A12).
342 * Other pins will return ADC_ERROR_VALUE.
343 * This function is interrupt safe, so it will restore the adc to the state it was before being called
344 * If more than one ADC exists, it will select the module with less workload, you can force a selection using
345 * adc_num. If you select ADC1 in Teensy 3.0 it will return ADC_ERROR_VALUE.
346 */
347 int ADC::analogReadDifferential(uint8_t pinP, uint8_t pinN, int8_t adc_num)
348 {
349 
350 #ifdef ADC_SINGLE_ADC
351  return adc0->analogReadDifferential(pinP, pinN); // use ADC0
352 #else
353  /* Teensy 3.1
354  */
355  if (adc_num == -1)
356  { // use no ADC in particular
357  // check which ADC can read the pin
358  bool adc0Pin = adc0->checkDifferentialPins(pinP, pinN);
359  bool adc1Pin = adc1->checkDifferentialPins(pinP, pinN);
360 
361  if (adc0Pin && adc1Pin)
362  { // Both ADCs
363  if ((adc0->num_measurements) > (adc1->num_measurements))
364  { // use the ADC with less workload
365  return adc1->analogReadDifferential(pinP, pinN);
366  }
367  else
368  {
369  return adc0->analogReadDifferential(pinP, pinN);
370  }
371  }
372  else if (adc0Pin)
373  { // ADC0
374  return adc0->analogReadDifferential(pinP, pinN);
375  }
376  else if (adc1Pin)
377  { // ADC1
378  return adc1->analogReadDifferential(pinP, pinN);
379  }
380  else
381  { // pins not valid in any ADC
382  adc0->fail_flag |= ADC_ERROR::WRONG_PIN;

384  return ADC_ERROR_VALUE; // all others are invalid
385  }
386  }

388  { // user wants ADC0

401 // Starts an analog measurement on the pin and enables interrupts.
402 /* It returns immediately, get value with readSingle().
403 * If the pin is incorrect it returns ADC_ERROR_VALUE
404 * This function is interrupt safe. The ADC interrupt will restore the adc to its previous settings and
405 * restart the adc if it stopped a measurement. If you modify the adc_isr then this won't happen.
406 */
407 bool ADC::startSingleRead(uint8_t pin, int8_t adc_num)
408 {
409 #ifdef ADC_SINGLE_ADC
410  return adc0->startSingleRead(pin); // use ADC0
411 #else
